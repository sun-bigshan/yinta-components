# Table 表格

用于展示多条结构类似的数据，可对数据进行排序、筛选、对比或其他自定义操作。

# 基本用法

基础的表格展示用法。

<template>
  <div class="table-box">
    <yinta-table :table-data="tableData" :table-column="tableColumn">
      <template #odoCode="{ row }">
        <el-link :underline="false" type="primary" @click="detailsDialog(row)">
          {{ row.odoCode }}
        </el-link>
      </template>
    </yinta-table>
  </div>
</template>

<script>
  export default {
    data() {
      return {
        tableColumn: [
          {
            prop: 'selection',
          },
          {
            label: '序号',
            prop: 'index',
            width: '50px',
          },
          {
            label: '仓库',
            prop: 'warehose',
          },
          {
            label: '出库单号',
            slot: 'odoCode',
            showTooltip: true,
          },
        ],
        tableData: [
          { warehose: 'USWA', odoCode: 123456 },
          { warehose: 'USWA', odoCode: 123456 },
          { warehose: 'USWA', odoCode: 123456 },
        ],
      };
    },
    methods: {
      detailsDialog(item) {
        console.log(item);
      },
    },
  };
</script>
